 73 /**
 74  * Action Game
 75  * Base class of enchant.Sprite
 76  * @scope enchant.ex.ExSprite.prototype
 77  */
 78 enchant.ex.ExSprite = enchant.Class.create(enchant.Sprite, {
 79     /**
 80      * @name enchant.ex.ExSprite
 81      * @class
 82      * Constructor of ExSprite
 83      * @param {Integer} width Spriteの横幅.
 84      * @param {Integer} height Spriteの高さ.
 85      * @constructs
 86      * @extends enchant.Sprite
 87      */
 88     initialize: function(width, height){
 89         enchant.Sprite.call(this, width, height);
 90 
 91         // Event arguments
 92         this._followArg;
 93         // collision
 94         this._collisionRect;
 95         this._isCollisionState = false;
 96         this._isCollision = false;
 97         this._collisionObjects = new Array();;
 98         this._collisionDuplicateObjects = new Array();
 99         // moved
100         this._moved = {x: 0, y: 0};
101         // history
102         this._history = {x: this.x, y: this.y};
103         // collision Based
104         this.COLLISION = {
105             INTERSECT_BASED: "intersect",
106             WITHIN_BASED: "within"
107         };
108         this._collisionBased = this.COLLISION.INTERSECT_BASED;
109         // forrow Based
110         this.FORROW = {
111             ABSOLUTE_BASED: "absolute",
112             RELATIVE_BASED: "relative"
113         };
114         this._forrowBased = this.FORROW.RELATIVE_BASED;
115 
116         // Event Added to scene
117         this.addEventListener(Event.ADDED_TO_SCENE, function(){
118             this._history.x = this.x;
119             this._history.y = this.y;
120         });
121 
122         // for follow
123         this.addEventListener(Event.ENTER_FRAME, function(){
124             this._moved.x = this.x - this._history.x;
125             this._moved.y = this.y - this._history.y;
126             this._history.x = this.x;
127             this._history.y = this.y;
128         });
129 
130         // judge collision Target
131         this.addEventListener(Event.ENTER_FRAME, function(){
132             this._isCollision = false;
133             // collision Sprite
134             if (this._collisionObjects.length > 0) {
135                 for (var i = 0; i < this._collisionObjects.length; i++) {
136                     (function(_this, value) {
137                         if (value instanceof Sprite && value._isContainedInCollection && !value.isCollisionIgnore || value instanceof Map) {
138                             _this._judgeCollision(value);
139                         } else if (value instanceof Array) {
140                             for (var i = 0; i < value.length; i++) {
141                                 arguments.callee(_this, value[i]);
142                             }
143                         } else if (value instanceof Group) {
144                             for (var i = 0; i < value.childNodes.length; i++) {
145                                 arguments.callee(_this, value.childNodes[i]);
146                             }
147                         }
148                     })(this, this._collisionObjects[i]);
149                 }
150             }
151         });
152 
153     },
154     /**
155      * 衝突判定の対象を設定します。
156      * @type enchant.Sprite | enchant.Group | Array
157      */
158     collision: {
159         set: function(value) {
160             this._collisionObjects = new Array();
161             this._collisionObjects.push(value);
162         }
163     },
164     /**
165      * 衝突の状態を取得します。
166      */
167     isCollision: {
168         get: function() {
169             return this._isCollision;
170         }
171     },
172     /**
173      * 衝突判定を行うSpriteを追加します。
174      * @param {enchant.Sprite | enchant.Group | Array} value 追加するSprite、またはそれを含むオブジェクト。
175      */
176     addCollision: function(value) {
177         this._collisionObjects.push(value);
178     },
179     removeCollision: function(value) {
180         if ((i = this._collisionObjects.indexOf(value)) !== -1) {
181             this._collisionObjects.splice(i, 1);
182         }
183     },
184     _addChildCollisionRect: function(sprite) {
185         // remove
186         this.addEventListener(Event.REMOVE_TO_SCENE, function(){
187             this.removeCollisionRect();
188         });
189         // add
190         if (this.parentNode) {
191             this.parentNode.addChild(sprite);
192         } else {
193             this.addEventListener(Event.ADDED_TO_SCENE, function(){
194                 this.parentNode.addChild(sprite);
195             });
196         }
197     },
198     _addCollisionRect: function(offsetX, offsetY) {
199         this.addEventListener(Event.ENTER_FRAME, function(){
200             this._collisionRect._followArg = arguments.callee;
201             this._collisionRect.x = this.x + offsetX;
202             this._collisionRect.y = this.y + offsetY;
203         });
204     },
205     removeCollisionRect: function() {
206         if (this._collisionRect) {
207             if (this._collisionRect._followArg) {
208                 this.removeEventListener(Event.ENTER_FRAME, this._collisionRect._followArg);
209                 this._collisionRect._followArg = null;
210             }
211             this._collisionRect.remove();
212         }
213     },
214     addCollisionRectScale: function(scaleX, scaleY) {
215         this.removeCollisionRect();
216         this._collisionRect = new Sprite(this.width * scaleX, this.height * scaleY);
217         this._collisionRect.isCollisionIgnore = true;
218         this._addChildCollisionRect(this._collisionRect);
219         this._addCollisionRect(this.width / 2 - this.width * scaleX / 2, this.height / 2 - this.height * scaleY / 2);
220     },
221     addCollisionRectSize: function(width, height, x, y) {
222         this.removeCollisionRect();
223         this._collisionRect = new Sprite(width, height);
224         this._collisionRect.isCollisionIgnore = true;
225         this._addChildCollisionRect(this._collisionRect);
226         this._addCollisionRect(x || this.width / 2 - width / 2, y || this.height / 2 - height / 2);
227     },
228     _judgeCollision: function(target) {
229         var result = false;
230         var thisRect = this._collisionRect ? this._collisionRect : this;
231         var targetRect = target._collisionRect ? target._collisionRect : target;
232         if (target instanceof Map) {
233             result = true;
234             target.collisionTile = null;
235             var _offsetX = targetRect._offsetX;
236             var _offsetY = targetRect._offsetY;
237             var _rect = thisRect.getOrientedBoundingRect();
238             _rect.center = [
239                 _rect.leftTop[0] + (_rect.rightTop[0] - _rect.leftTop[0]) / 2,
240                 _rect.leftTop[1] + (_rect.leftBottom[1] - _rect.leftTop[1]) / 2
241             ];
242             // Center
243             if (targetRect.hitTest(_rect.center[0] - _offsetX, _rect.center[1] - _offsetY)) {
244                 target.collisionTile = targetRect.checkTile(_rect.leftTop[0] - _offsetX, _rect.leftTop[1] - _offsetY);
245             // TopLeft
246             } else if (targetRect.hitTest(_rect.leftTop[0] - _offsetX, _rect.leftTop[1] - _offsetY)) {
247                 target.collisionTile = targetRect.checkTile(_rect.leftTop[0] - _offsetX, _rect.leftTop[1] - _offsetY);
248             // TopRight
249             } else if (targetRect.hitTest(_rect.rightTop[0] - _offsetX, _rect.rightTop[1] - _offsetY)) {
250                 target.collisionTile = targetRect.checkTile(thisRect._offsetX + thisRect.width - _offsetX, thisRect._offsetY - _offsetY);
251             // LeftBottom
252             } else if (targetRect.hitTest(_rect.leftBottom[0] - _offsetX, _rect.leftBottom[1] - _offsetY)) {
253                 target.collisionTile = targetRect.checkTile(thisRect._offsetX - _offsetX, thisRect._offsetY + thisRect.height - _offsetY);
254             // RightBottom
255             } else if (targetRect.hitTest(_rect.rightBottom[0] - _offsetX, _rect.rightBottom[1] - _offsetY)) {
256                 target.collisionTile = targetRect.checkTile(thisRect._offsetX + thisRect.width - _offsetX, thisRect._offsetY + thisRect.height - _offsetY);
257             } else {
258                 result = false;
259             }
260         } else {
261             if (this._collisionBased == this.COLLISION.INTERSECT_BASED) {
262                 result = thisRect.intersectStrict(targetRect);
263             } else {
264                 result = thisRect.within(targetRect, (thisRect.width + thisRect.height) / 4 + (targetRect.width + targetRect.height) / 4);
265             }
266         }
267         if (result) {
268             this._isCollision = true;
269             target._isCollisionState = true;
270             this._dispatchEventCollision(target);
271             if (this._moved.x < 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_TO_LEFT);
272             if (this._moved.x > 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_TO_RIGHT);
273             if (this._moved.y < 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_TO_TOP);
274             if (this._moved.y > 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_TO_BOTTOM);
275             if (target._moved) {
276                 if (target._moved.x > 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_FROM_LEFT);
277                 if (target._moved.x < 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_FROM_RIGHT);
278                 if (target._moved.y > 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_FROM_TOP);
279                 if (target._moved.y < 0) this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_FROM_BOTTOM);
280             }
281             return true;
282         }
283         target._isCollisionState = false;
284         return false;
285     },
286     _dispatchEventCollision: function(target) {
287         var e;
288         var existCount = this._collisionDuplicateObjects.indexOf(target);
289         if (existCount < 0) {
290             this._dispatchEventMakeCollision(target, enchant.Event.COLLISION);
291             this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_START);
292             this._collisionDuplicateObjects.push(target);
293             //ターゲットとの衝突がなくなったときイベント生成
294             this.addEventListener(Event.ENTER_FRAME, function() {
295                 var _arg = arguments.callee;
296                 if (!target._isCollisionState) {
297                     this._collisionDuplicateObjects.splice(existCount);
298                     this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_END);
299                     this.removeEventListener(Event.ENTER_FRAME, _arg);
300                 }
301             });
302         }
303         this._dispatchEventMakeCollision(target, enchant.Event.COLLISION_TICK);
304     },
305     _dispatchEventMakeCollision: function(target, collisionName) {
306         e = new Event(collisionName);
307         e.collision = {};
308         e.collision.target = target;
309         e.collisionTarget = target;
310         this.dispatchEvent(e);
311     },
312     /**
313      * ターゲットに指定したSpriteと一緒に移動します。
314      * @param {enchant.Sprite} target ターゲットを指定します。
315      * @example
316      * var target = new ExSprite(32, 32);
317      * core.rootScene.addChild(target);
318      *
319      * var sprite = new ExSprite(32, 32);
320      * sprite.follow(target);
321      * core.rootScene.addChild(sprite);
322      *
323      * target.tl.moveTo(10, 10, 10);
324      * //sprite.tl.moveTo(10, 10, 10);
325      *
326      */
327     follow: function(target) {
328         this.unfollow();
329         if (this._forrowBased == this.FORROW.RELATIVE_BASED) {
330             this.addEventListener(Event.ENTER_FRAME, function() {
331                 this._followArg = arguments.callee;
332                 this.x += target._moved.x;
333                 this.y += target._moved.y;
334             });
335         } else {
336             this.addEventListener(Event.ENTER_FRAME, function() {
337                 this._followArg = arguments.callee;
338                 this.x += target._moved.x;
339                 this.y += target._moved.y;
340             });
341         }
342     },
343     /**
344      * followを解除します。
345      */
346     unfollow: function() {
347         if (this._followArg) {
348             this.removeEventListener(Event.ENTER_FRAME, this._followArg);
349             this._followArg = null;
350         }
351     },
352     setForrowRelativeBased: function() {
353         this._forrowBased = this.FORROW.RELATIVE_BASED;
354     },
355     setForrowAbsoluteBased: function() {
356         this._forrowBased = this.FORROW.ABSOLUTE_BASED;
357     },
358     setCollisionIntersectBased: function() {
359         this._collisionBased = this.COLLISION.INTERSECT_BASED;
360     },
361     setCollisionWithinBased: function() {
362         this._collisionBased = this.COLLISION.WITHIN_BASED;
363     }
364 });
365 /**
366  * 指定オブジェクト内でx方向の中央寄せを行う。
367  * @param {Object} [another] 基準となるオブジェクト。（省略時は親Nodeとなる）
368  */
369 enchant.ex.ExSprite.prototype.alignHorizontalCenterIn = function(another) {
370     var parentNode = this.parentNode;
371     if (another) parentNode = another;
372     if (parentNode) {
373         this.x = ~~(parentNode.width / 2) - ~~(this.width / 2);
374     }
375 };
376 /**
377  * 指定オブジェクト内でy方向の中央寄せを行う。
378  * @param {Object} [another] 基準となるオブジェクト。（省略時は親Nodeとなる）
379  */
380 enchant.ex.ExSprite.prototype.alignVerticalCenterIn = function(another) {
381     var parentNode = this.parentNode;
382     if (another) parentNode = another;
383     if (parentNode) {
384         this.y = ~~(parentNode.height / 2) - ~~(this.height / 2);
385     }
386 };
